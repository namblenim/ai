    <script>
        const world = document.getElementById('world');
        const root = document.getElementById('robot-root');
        const visual = document.getElementById('robot-visual');
        const hand = document.getElementById('robot-hand');
        const armLine = document.getElementById('robot-arm-line');
        const statusText = document.getElementById('status-text');
        const posYText = document.getElementById('pos-y');
        const eye = document.getElementById('robot-eye');
        const goal = document.getElementById('goal');

        // 월드 좌표계
        let worldX = 1300;  // 월드 좌표
        let worldY = 160;   // 높이
        let isMoving = false;
        let isReaching = false;
        let currentKey = null;
        let currentReachDist = 0;

        const MAX_DIST = 150;
        const REACH_SPEED = 6.0;
        const GRAVITY = 0.3;
        const MAX_FALL_SPEED = 8;
        const SLOPE_ANGLE = -26.5; // 도(degrees)

        // 지형 정의 (월드 좌표)
        const PLATFORM_LEFT = { x: 0, y: 300, w: 300, h: 80 };        // 높은 곳
        const SLOPE = { x: 300, y: 80, w: 800, h: 300 };             // 경사면
        const PLATFORM_RIGHT = { x: 1100, y: 80, w: 900, h: 80 };    // 낮은 곳
        const GOAL = { x: 1150, y: 180 };

        let velocityY = 0;
        let onSlope = false;
        let angularRotation = 0;

        function setStatus(text, type = '') {
            statusText.innerText = text;
            if (type === 'active') statusText.className = 'status-active';
            else if (type === 'reaching') statusText.className = 'status-reaching';
            else if (type === 'warning') statusText.className = 'status-warning';
            else statusText.className = '';

            if (type === 'reaching') eye.style.boxShadow = '0 0 15px #ffcc00';
            else if (type === 'warning') eye.style.boxShadow = '0 0 15px #ff4757';
            else eye.style.boxShadow = '0 0 10px #00f2ff';
        }

        // 점이 경사면 위에 있는지 확인
        function isOnSlope(x, y) {
            if (x < SLOPE.x || x > SLOPE.x + SLOPE.w) return false;

            // 경사면: (300, 380) ~ (1100, 80)
            const slope = (SLOPE.y - (SLOPE.y + SLOPE.h)) / SLOPE.w;
            const slopeY = (SLOPE.y + SLOPE.h) + slope * (x - SLOPE.x);
            
            return Math.abs(y - slopeY) < 15; // 오차범위
        }

        function getHeightOnSlope(x) {
            const slope = (SLOPE.y - (SLOPE.y + SLOPE.h)) / SLOPE.w;
            return (SLOPE.y + SLOPE.h) + slope * (x - SLOPE.x);
        }

        // 카메라 업데이트
        function updateCamera() {
            const targetCameraX = Math.max(0, worldX - 450); // 화면 중심에 로봇 유지
            world.style.transform = `translateX(${-targetCameraX}px)`;
        }

        // 중력 및 경사면 물리
        function applyPhysics() {
            const slopeHeight = getHeightOnSlope(worldX);
            
            if (worldY >= slopeHeight - 5) {
                // 경사면 위에 있음
                onSlope = true;
                worldY = slopeHeight;
                velocityY = 0;

                // 경사면에서는 중력에 의해 오른쪽으로 계속 밀림 (자동 미끄러짐)
                if (!isMoving && !isReaching && worldX < PLATFORM_RIGHT.x + PLATFORM_RIGHT.w - 80) {
                    worldX += 1.5; // 자동 미끄러짐
                }

                // 기울기 보정
                angularRotation = SLOPE_ANGLE;
            } else if (worldY > PLATFORM_RIGHT.y) {
                // 오른쪽 평지 위
                onSlope = false;
                worldY = PLATFORM_RIGHT.y;
                velocityY = 0;
                angularRotation = 0;
            } else if (worldY > PLATFORM_LEFT.y) {
                // 왼쪽 평지 위
                onSlope = false;
                worldY = PLATFORM_LEFT.y;
                velocityY = 0;
                angularRotation = 0;
            } else {
                // 공중에 떨어짐
                onSlope = false;
                velocityY += GRAVITY;
                if (velocityY > MAX_FALL_SPEED) velocityY = MAX_FALL_SPEED;
                worldY += velocityY;
                angularRotation = 0;

                // 맵 아래로 떨어졌으면 리셋
                if (worldY > 600) {
                    resetPosition();
                }
            }

            updateCamera();
            posYText.innerText = Math.round(worldY);
        }

        function resetPosition() {
            worldX = 1300;
            worldY = 160;
            velocityY = 0;
            setStatus("위치 초기화됨", "warning");
        }

        function updateReach() {
            if (!isReaching) return;

            const dir = currentKey === 'd' ? 1 : -1;
            let isBlocked = false;

            let handWorldX = worldX + (dir === 1 ? 40 + currentReachDist + 20 : 40 - currentReachDist);

            // 경계 체크
            if (handWorldX <= 0 || handWorldX >= 2000) {
                isBlocked = true;
            }

            if (currentReachDist < MAX_DIST && !isBlocked) {
                currentReachDist += REACH_SPEED;
                if (currentReachDist > MAX_DIST) currentReachDist = MAX_DIST;

                const armTargetPos = 40 + (dir * currentReachDist);
                hand.style.left = `${armTargetPos}px`;

                if (dir === 1) {
                    armLine.style.left = `40px`;
                    armLine.style.width = `${currentReachDist}px`;
                } else {
                    armLine.style.left = `${armTargetPos + 8}px`;
                    armLine.style.width = `${currentReachDist}px`;
                }
                requestAnimationFrame(updateReach);
            } else if (isBlocked) {
                setStatus("벽에 막혔습니다", "warning");
            }
        }

        async function pullBody() {
            if (currentReachDist <= 5) {
                resetArm();
                return;
            }

            isMoving = true;
            const dir = currentKey === 'd' ? 1 : -1;

            const maxDuration = 600;
            const minDuration = 400;
            const stretchRatio = currentReachDist / MAX_DIST;
            const pullDuration = maxDuration - (stretchRatio * (maxDuration - minDuration));

            setStatus("이동 중...", "active");

            let targetX = worldX + (dir * currentReachDist);
            targetX = Math.max(0, Math.min(targetX, 2000));

            worldX = targetX;
            
            root.style.transition = `left ${pullDuration}ms cubic-bezier(0.42, 0, 0.58, 1)`;
            root.style.left = `${worldX - (worldX - 450 >= 0 ? Math.max(0, worldX - 450) : 0)}px`;

            hand.style.transition = `left ${pullDuration}ms ease-in-out`;
            armLine.style.transition = `width ${pullDuration}ms ease-in-out, left ${pullDuration}ms ease-in-out`;

            hand.style.left = `40px`;
            armLine.style.width = `0px`;
            armLine.style.left = `40px`;

            await new Promise(r => setTimeout(r, pullDuration));

            resetArm();
            isMoving = false;
            root.style.transition = "none";
            currentReachDist = 0;

            setStatus("대기 중");

            // 목표 도달 확인
            if (worldX > GOAL.x - 40 && worldX < GOAL.x + 60 && worldY < GOAL.y + 80) {
                setStatus("🎉 성공! 경사로를 올라갔습니다!", "active");
            }
        }

        function resetArm() {
            hand.style.transition = "none";
            armLine.style.transition = "none";
            hand.style.left = `40px`;
            armLine.style.width = `0px`;
            armLine.style.left = `40px`;
        }

        function updateRobotPosition() {
            const cameraPosX = Math.max(0, worldX - 450);
            const screenX = worldX - cameraPosX;
            root.style.left = `${screenX}px`;
            root.style.bottom = `${worldY + 80}px`;
            visual.style.transform = `scaleX(1) rotateZ(${angularRotation}deg)`;
        }

        window.addEventListener('keydown', (e) => {
            const key = e.key.toLowerCase();
            if ((key === 'a' || key === 'd') && !isMoving && !isReaching) {
                isReaching = true;
                currentKey = key;
                currentReachDist = 0;
                visual.style.transform = key === 'd' ? `scaleX(1) rotateZ(${angularRotation}deg)` : `scaleX(-1) rotateZ(${angularRotation}deg)`;
                setStatus("팔 뻗는 중...", "reaching");
                updateReach();
            }
        });

        window.addEventListener('keyup', (e) => {
            const key = e.key.toLowerCase();
            if (key === currentKey && isReaching) {
                isReaching = false;
                pullBody();
                currentKey = null;
            }
        });

        // 메인 루프
        let lastFrameTime = Date.now();
        function gameLoop() {
            applyPhysics();
            updateRobotPosition();
            requestAnimationFrame(gameLoop);
        }

        gameLoop();
    </script>
