    <script>
        const root = document.getElementById('robot-root');
        const visual = document.getElementById('robot-visual');
        const hand = document.getElementById('robot-hand');
        const armLine = document.getElementById('robot-arm-line');
        const statusText = document.getElementById('status-text');
        const eye = document.getElementById('robot-eye');
        
        const wall = document.getElementById('falling-wall');
        const plate = document.getElementById('pressure-plate');
        const gate = document.getElementById('gate');
        const gateLight = document.getElementById('gate-light');

        let currentX = 300; 
        let isMoving = false;
        let isReaching = false;
        let currentKey = null;
        let currentReachDist = 0;
        
        let isWallFallen = false;
        let isGateOpen = false;

        const MAX_DIST = 150; 
        const REACH_SPEED = 6.0;

        // 지형 좌표 정보
        const GATE_X = 30;
        const WALL_X = 600;

        function setStatus(text, type = '') {
            statusText.innerText = text;
            if (type === 'active') statusText.className = 'status-active';
            else if (type === 'reaching') statusText.className = 'status-reaching';
            else if (type === 'collision') statusText.className = 'status-collision';
            else statusText.className = '';

            if (type === 'reaching') eye.style.boxShadow = '0 0 15px #ffcc00';
            else if (type === 'collision') eye.style.boxShadow = '0 0 15px #ff4757';
            else eye.style.boxShadow = '0 0 10px #00f2ff';
        }

        function updateReach() {
            if (!isReaching) return;

            const dir = currentKey === 'd' ? 1 : -1;
            let isBlocked = false;
            
            // 손의 월드 좌표 계산
            let currentHandWorldX = currentX + (dir === 1 ? 40 + currentReachDist + 20 : 40 - currentReachDist);

            if (dir === 1) {
                // 오른쪽으로 뻗을 때 벽 충돌 체크
                if (!isWallFallen && currentHandWorldX >= WALL_X) isBlocked = true;
                if (currentHandWorldX >= 900) isBlocked = true;
            } else {
                // 왼쪽으로 뻗을 때 문 충돌 체크
                if (!isGateOpen && currentHandWorldX <= GATE_X + 30) isBlocked = true;
                if (currentHandWorldX <= 0) isBlocked = true;
            }

            if (currentReachDist < MAX_DIST && !isBlocked) {
                currentReachDist += REACH_SPEED;
                if (currentReachDist > MAX_DIST) currentReachDist = MAX_DIST;
                
                const armTargetPos = 40 + (dir * currentReachDist);
                hand.style.left = `${armTargetPos}px`;
                
                if (dir === 1) {
                    armLine.style.left = `40px`;
                    armLine.style.width = `${currentReachDist}px`;
                } else {
                    armLine.style.left = `${armTargetPos + 8}px`;
                    armLine.style.width = `${currentReachDist}px`;
                }
                requestAnimationFrame(updateReach);
            } else if (isBlocked) {
                setStatus("차단됨", "collision");
            }
        }

        async function pullBody() {
            if (currentReachDist <= 5) {
                resetArm();
                return;
            }

            isMoving = true;
            const dir = currentKey === 'd' ? 1 : -1;
            
            const maxDuration = 600; 
            const minDuration = 400; 
            const stretchRatio = currentReachDist / MAX_DIST;
            const pullDuration = maxDuration - (stretchRatio * (maxDuration - minDuration));

            setStatus("이동 중...", "active");
            
            let targetX = currentX + (dir * currentReachDist);
            let collisionOccurred = false;
            let weakImpact = false;

            if (dir === 1) {
                // 오른쪽 본체 충돌 감지
                const bodyRightX = targetX + 80; 
                if (!isWallFallen && bodyRightX >= WALL_X) {
                    targetX = WALL_X - 80;
                    if (currentReachDist >= MAX_DIST - 5) {
                        collisionOccurred = true;
                    } else {
                        weakImpact = true;
                    }
                }
            } else {
                // 왼쪽 본체 충돌 감지 (닫힌 문)
                const bodyLeftX = targetX;
                if (!isGateOpen && bodyLeftX <= GATE_X + 30) {
                    targetX = GATE_X + 30;
                }
            }
            
            currentX = Math.max(0, Math.min(targetX, 820));
            
            root.style.transition = `left ${pullDuration}ms cubic-bezier(0.42, 0, 0.58, 1)`;
            root.style.left = `${currentX}px`;

            hand.style.transition = `left ${pullDuration}ms ease-in-out`;
            armLine.style.transition = `width ${pullDuration}ms ease-in-out, left ${pullDuration}ms ease-in-out`;
            
            hand.style.left = `40px`;
            armLine.style.width = `0px`;
            armLine.style.left = `40px`;

            await new Promise(r => setTimeout(r, pullDuration));

            if (collisionOccurred) {
                triggerWallFall();
            } else if (weakImpact) {
                setStatus("충격 부족!", "collision");
            } else {
                setStatus("대기 중");
            }

            resetArm();
            isMoving = false;
            root.style.transition = "none";
            currentReachDist = 0;
        }

        function triggerWallFall() {
            isWallFallen = true;
            wall.style.transform = "rotate(90deg)"; // 오른쪽으로 회전
            setStatus("벽 파괴!", "active");
            
            setTimeout(() => {
                plate.style.height = "3px";
                plate.style.background = "#27ae60";
                openGate();
            }, 800);
        }

        function openGate() {
            isGateOpen = true;
            gate.style.bottom = "230px";
            gateLight.style.background = "#2ecc71";
            setStatus("문 열림", "active");
        }

        function resetArm() {
            hand.style.transition = "none";
            armLine.style.transition = "none";
            hand.style.left = `40px`;
            armLine.style.width = `0px`;
            armLine.style.left = `40px`;
        }

        window.addEventListener('keydown', (e) => {
            const key = e.key.toLowerCase();
            if ((key === 'a' || key === 'd') && !isMoving && !isReaching) {
                isReaching = true;
                currentKey = key;
                currentReachDist = 0;
                visual.style.transform = key === 'd' ? 'scaleX(1)' : 'scaleX(-1)';
                setStatus("팔 뻗는 중...", "reaching");
                updateReach();
            }
        });

        window.addEventListener('keyup', (e) => {
            const key = e.key.toLowerCase();
            if (key === currentKey && isReaching) {
                isReaching = false;
                pullBody();
                currentKey = null;
            }
        });

        root.style.left = `${currentX}px`;
    </script>
